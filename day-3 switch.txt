#include <stdio.h>
int main() {
    int choice;
    printf("Choose an option:\n");
    printf("1. Tamil\n2. English\n3. Maths\n4. Exit\n");
    scanf("%d", &choice);
    switch (choice) {
        case 1:
            printf("You selected Tamil.\n");
            break;
        case 2:
            printf("You selected English.\n");
            break;
        case 3:
            printf("You selected Maths.\n");
            break;
        case 4:
            printf("Exiting program.\n");
            break;
        default:
            printf("Invalid choice. Try again.\n");
    }
    return 0;
}
